db.createCollection("systemTypes", {
   validator: {
      $jsonSchema: {
         bsonType: "object",
         required: ["name"],
         properties: {
            name: {
               enum: [ "engine", "fuel", "hydraulic" ],
               description: "can only be one of the enum values and is required"
            },
            description: {
               bsonType: "string",
               description: "must be a string"
			}
		}
	  }
	}
})
